      - name: validate deb contents
        run: |
          DEB_FILE=$(ls target/release/bundle/deb/robot-controller_*.deb 2>/dev/null | head -1)
          echo "=== dpkg -c on $DEB_FILE ==="
          dpkg -c "$DEB_FILE"

          dpkg -c "$DEB_FILE" | grep "usr/share/applications/robot-controller.desktop" || { echo "ERROR: .desktop file not in deb"; exit 1; }
