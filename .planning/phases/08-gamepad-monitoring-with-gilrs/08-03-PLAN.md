       ```rust
       fn get_direction_from_axes(x: f32, y: f32) -> Direction {
           let abs_x = x.abs();
           let abs_y = y.abs();
           
           // Deadzone check first (D-15, D-16)
           if abs_x < DEADZONE && abs_y < DEADZONE {
               return Direction::S;
           }
           
           // If |y| > |x| then F/B, else L/R (D-16)
           if abs_y > abs_x {
               if y < 0.0 {
                   Direction::F // Forward
               } else {
                   Direction::B // Backward
               }
           } else {
               if x < 0.0 {
                   Direction::L // Left
               } else {
                   Direction::R // Right
               }
           }
       }
       ```
